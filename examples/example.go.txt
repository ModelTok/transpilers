package main

func add(a int64, b int64) int64 {
    return a + b
}

func factorial(n int64) int64 {
    var result int64 = 1
    var i int64 = 1
    for i <= n {
        result = result * i
        i = i + 1
    }
    return result
}

func sumTo(n int64) int64 {
    var total int64 = 0
    for i := int64(0); i < n; i++ {
        total = total + i
    }
    return total
}
